#include<iostream>
#include"cmath"
using namespace std;


float diskr(float a, float b, float c)
{
        float x = b*b - 4 * a*c;
        return x;
}
float perv(float a, float b, float d)
{
        float x = (-b+sqrt(d))/(2*a);
        return x;
}
float vtor(float a, float b, float d)
{
        float x = (-b - sqrt(d)) / (2 * a);
        return x;
}

{
        float a, b, c, d;
